This inspection reports when the SInterpolator#interpolate function is used manually. 


    // incorrect
    class Foo {
        void foo() {
            var name = "Alex";
            var result = new SInterpolator("hello, ", "").interpolate(name);
        }
    }



    // correct
    class Foo {
        void foo() {
            var name = "Alex";
            var result = s("hello, $name");
        }
    }
