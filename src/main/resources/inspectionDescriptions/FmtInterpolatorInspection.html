This inspection reports when the Interpolations#fmt function is used with wrong string value. Only string literal value or concatenation of string literals are supported at this place. 


    // incorrect
    class Foo {
        void foo() {
            var result = Interpolations.fmt(null);
        }
    }

    // incorrect
    class Foo {
        void foo() {
            var tmp = "template";
            var result = Interpolations.fmt(tmp);
        }
    }



    // correct
    class Foo {
        void foo() {
            var name = "Alex";
            var result = Interpolations.fmt("Hello %s${name}");
        }
    }

    // correct
    class Foo {
        void foo() {
            var result = Interpolations.fmt("hello, " + "%s$name");
        }
    }
