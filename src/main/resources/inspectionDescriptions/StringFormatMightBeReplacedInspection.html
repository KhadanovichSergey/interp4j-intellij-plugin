This inspection reports when the String#format is used in places where string interpolation can be used. 


    // incorrect
    class Foo {
        void foo() {
            var name = "Alex";
            var greet = String.format("Hello, %s", name);
        }
    }



    // correct
    class Foo {
        void foo() {
            var name = "Alex";
            var result = s("hello, $name");
        }
    }
