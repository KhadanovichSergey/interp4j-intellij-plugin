This inspection reports when the Interpolations#s function is used with wrong string value. Only string literal value or concatenation of string literals are supported at this place. 


    // incorrect
    class Foo {
        void foo() {
            var result = Interpolations.s(null);
        }
    }

    // incorrect
    class Foo {
        void foo() {
            var tmp = "template";
            var result = Interpolations.s(tmp);
        }
    }



    // correct
    class Foo {
        void foo() {
            var name = "Alex";
            var result = Interpolations.s("Hello ${name}");
        }
    }

    // correct
    class Foo {
        void foo() {
            var result = Interpolations.s("hello, " + "$name");
        }
    }
