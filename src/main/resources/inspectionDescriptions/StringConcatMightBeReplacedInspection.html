This inspection reports when the string concatenation is used in places where string interpolation can be used. 


    // incorrect
    class Foo {
        void foo() {
            var name = "Alex";
            var greet = "Hello, " + name;
        }
    }



    // correct
    class Foo {
        void foo() {
            var name = "Alex";
            var result = s("hello, $name");
        }
    }
